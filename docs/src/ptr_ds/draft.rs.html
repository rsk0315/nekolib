4//!
5//! ### 生ポインタの使用
6//!
7//! 構造体 `Foo` と、それを参照する構造体 `FooRef<BorrowType>` を考える。
8//! `BorrowType` は今後追加していくが、一旦 `Boxed<Foo>` のように振る舞う `marker::Owned`
9//! と、`&'a Foo` のように振る舞う `marker::Immut<'a>` を考える。
10//!
11//! ```
12//! use std::{marker::PhantomData, ptr::NonNull};
13//!
14//! struct Foo(u32);
15//! struct FooRef<BorrowType> {
16//!     foo: NonNull<Foo>,
17//!     _marker: PhantomData<BorrowType>,
18//! }
19//!
20//! mod marker {
21//!     use std::marker::PhantomData;
22//!
23//!     pub enum Owned {}
24//!     pub struct Immut<'a>(PhantomData<&'a ()>);
25//! }
26//!
27//! impl Foo {
28//!     pub fn new() -> Box<Self> { Box::new(Self(0)) }
29//! }
30//! impl FooRef<marker::Owned> {
31//!     pub fn new_ref() -> Self {
32//!         Self { foo: NonNull::from(Box::leak(Foo::new())), _marker: PhantomData }
33//!     }
34//! }
35//! impl<BorrowType> FooRef<BorrowType> {
36//!     pub fn borrow(&self) -> FooRef<marker::Immut<'_>> {
37//!         FooRef { foo: self.foo, _marker: PhantomData }
38//!     }
39//!     pub fn get(&self) -> &u32 { unsafe { &(*self.foo.as_ptr()).0 } }
40//! }
41//!
42//! let foo_ref = FooRef::new_ref();
43//! assert_eq!(*foo_ref.get(), 0);
44//!
45//! let foo_ref_immut = foo_ref.borrow();
46//! assert_eq!(*foo_ref_immut.get(), 0);
47//!
48//! assert_eq!(*foo_ref.get(), 0);
49//! assert_eq!(*foo_ref_immut.get(), 0);
50//!
51//! unsafe { drop(Box::from_raw(foo_ref.foo.as_ptr())) };
52//! ```
53//!
54//! 当然、`Immut<'a>` だけでなく `Mut<'a>` も欲しい。
55//! `drop` に関しては、`Boxed<Foo>` のように振る舞うところの `FooRef<marker::Owned>`
56//! でのみ実装したいが、そういうことはできない ([E0366])。
57//! ここでは、手動で呼び出すための `.drop()` を提供しつつ `Boxed<Foo>` のように振る舞う
58//! `FooRef<marker::Dying>` を別途作ることにする。
59//!
60//! また、連想配列を実装することを見据えると、一部のメンバ変数についてのみ可変参照を公開したいこともある。
61//! ここでは `struct Foo(u32, u32);` の `foo: Foo` に対して、`&foo.0` と `&mut foo.1`
62//! を公開する `FooRef<marker::SndMut<'a>>` を作ることにする。
63//! このとき、`&foo.0` にアクセスしても `&mut foo.1` が invalidate
64//! されないように気をつける必要がある（逆も然り）。
65//!
66//! [E0366]: https://doc.rust-lang.org/error_codes/E0366.html
67//!
68//! 加えて、`marker::Mut<'a>` の lifetime を消して、一時的に静的解析の対象外にできる
69//! `marker::DormantMut` も作っておく[^dormant]。もちろん、冒頭の例で触れたように
70//! Stacked Borrows のルールには従う必要がある。
71//!
72//! [^dormant]: dormant や (re-)awaken という表現がしばしば使われている印象がある。
73//!
74//! 別の marker を返すようなメソッドにおいては、`FooRef { foo: self.foo, _marker: PhantomData }`
75//! の boilerplate を都度書く必要があってややうれしくない。`PhantomData`
76//! の型パラメータを陽に書く必要がないので、見かけ上は全く同じものになっている。
77//! また、どの marker からどの marker への遷移をできるかを意識しておくとよいかもしれない。
78//!
79//! 他にも、`FooRef<marker::Immut<'a>>` は `Copy`/`Clone`
80//! であるとか、諸々の変換などの実装が欲しくなるであろう。
81//!
82//! ### 参照の無効化
83//!
84//! さて、`marker::Mut<'a>` で `.get_mut()` のようなメソッドを公開するにあたり、
85//! 無効な参照を返さないように気をつける必要がある。
86//! 最初の例のように、下記のコードは未定義動作となる。
87//!
88//! ```
89//! struct Foo(u32);
90//! let mut foo = Foo(0);
91//! ```
92//!
93//! ```ignore
94//! # struct Foo(u32);
95//! # let mut foo = Foo(0);
96//! let foo_ptr_1: *mut _ = &mut foo;
97//! let foo_ptr_2: *mut _ = &mut foo;
98//! unsafe { (*foo_ptr_1).0 }; // UB
99//! ```
100//!
101//! [`std::ptr`] には「`&mut foo` を `*mut _` にキャストするときは他の参照があってはいけない」とあり、
102//! `foo_ptr_1` が生きている状態で `&mut foo` しているのが悪いように読める。Stacked Borrows
103//! のルール的には、二つ目の `&mut foo` をした時点で `foo_ptr_1` が無効化されるので、
104//! その後の参照が未定義動作になるという説明になると思われる。
105//! 同じ可変参照から作ったポインタであれば問題ないようである。
106//!
107//! ```
108//! # struct Foo(u32);
109//! # let mut foo = Foo(0);
110//! let mut foo_mut = &mut foo;
111//! let foo_ptr_1: *mut _ = foo_mut;
112//! let foo_ptr_2: *mut _ = foo_mut;
113//! unsafe { (*foo_ptr_1).0 = 10 };
114//! unsafe { (*foo_ptr_2).0 = 20 };
115//! assert_eq!(foo.0, 20);
116//! ```
117//!
118//! [`NonNull`][`std::ptr::NonNull`] を介して生ポインタを作っても、[`.as_mut()`][`std::ptr::NonNull::as_mut`]
119//! を使って可変参照を作ってしまうとうまくいかない。
120//!
121//! ```ignore
122//! use std::ptr::NonNull;
123//!
124//! # struct Foo(u32);
125//! # let mut foo = Foo(0);
126//! let mut foo_nonnull = NonNull::from(&mut foo);
127//! let foo0_mut_1 = unsafe { &mut foo_nonnull.as_mut().0 };
128//! let foo0_mut_2 = unsafe { &mut foo_nonnull.as_mut().0 };
129//! *foo0_mut_1 = 10; // UB
130//! ```
131//!
132//! ```ignore
133//! use std::ptr::NonNull;
134//!
135//! # struct Foo(u32);
136//! # let mut foo = Foo(0);
137//! let mut foo_nonnull = NonNull::from(&mut foo);
138//! let foo_mut_1 = unsafe { foo_nonnull.as_mut() };
139//! let foo_mut_2 = unsafe { foo_nonnull.as_mut() };
140//! foo_mut_1.0 = 10; // UB
141//! ```
142//!
143//! 下記も同様。
144//!
145//! ```ignore
146//! use std::ptr::NonNull;
147//!
148//! # struct Foo(u32);
149//! # let mut foo = Foo(0);
150//! let mut foo_nonnull = NonNull::from(&mut foo);
151//! let foo0_mut_1 = unsafe { &mut (*foo_nonnull.as_ptr()).0 };
152//! let foo0_mut_2 = unsafe { &mut (*foo_nonnull.as_ptr()).0 };
153//! *foo0_mut_1 = 10; // UB
154//! ```
155//!
156//! 今度は、メンバを複数持つような場合を考える。
157//!
158//! ```
159//! struct Foo(u32, u32);
160//! let mut foo = Foo(0, 0);
161//! ```
162//!
163//! ```
164//! # struct Foo(u32, u32);
165//! # let mut foo = Foo(0, 0);
166//! let foo0_mut = &mut foo.0;
167//! let foo1_mut = &mut foo.1;
168//! *foo0_mut = 10;
169//! *foo1_mut = 20;
170//! assert_eq!((foo.0, foo.1), (10, 20)); // ok
171//! ```
172//!
173//! Safe Rust では上記のように書けるが、事情があって生ポインタを使う必要がある状況とする。
174//!
175//! ```
176//! use std::ptr::NonNull;
177//!
178//! # struct Foo(u32, u32);
179//! # let mut foo = Foo(0, 0);
180//! let foo_nonnull = NonNull::from(&mut foo);
181//! let foo0_mut = unsafe { &mut (*foo_nonnull.as_ptr()).0 };
182//! let foo1_mut = unsafe { &mut (*foo_nonnull.as_ptr()).1 };
183//! *foo0_mut = 10;
184//! *foo1_mut = 20;
185//! assert_eq!((foo.0, foo.1), (10, 20)); // ok
186//! ```
187//!
188//! 下記のように [`std::ptr::NonNull::as_mut`] を使うと、アクセスしていない部分も無効化されるように見える。
189//!
190//! ```ignore
191//! use std::ptr::NonNull;
192//!
193//! # struct Foo(u32, u32);
194//! # let mut foo = Foo(0, 0);
195//! let mut foo_nonnull = NonNull::from(&mut foo);
196//! let foo0_mut = unsafe { &mut foo_nonnull.as_mut().0 };
197//! let foo1_mut = unsafe { &mut foo_nonnull.as_mut().1 };
198//! *foo0_mut = 10; // UB
199//! ```
200//!
201//! [`std::ptr::NonNull::as_ptr`] を使うと無効化されない模様。
202//!
203//! ```
204//! use std::ptr::{addr_of_mut, NonNull};
205//!
206//! # struct Foo(u32, u32);
207//! # let mut foo = Foo(0, 0);
208//! let mut foo_nonnull = NonNull::from(&mut foo);
209//! let foo0_mut = unsafe { &mut *addr_of_mut!((*foo_nonnull.as_ptr()).0) };
210//! let foo1_mut = unsafe { &mut *addr_of_mut!((*foo_nonnull.as_ptr()).1) };
211//! *foo0_mut = 10;
212//! *foo1_mut = 20;
213//! assert_eq!((foo.0, foo.1), (10, 20)); // ok
214//! ```
215//!
216//! [`std::ptr::addr_of_mut`] を使わなくても問題ない模様 (cf. [`&`/`&mut`](https://doc.rust-lang.org/reference/expressions/operator-expr.html#borrow-operators), [`*`](https://doc.rust-lang.org/reference/expressions/operator-expr.html#the-dereference-operator), [place expressions](https://doc.rust-lang.org/reference/expressions.html#place-expressions-and-value-expressions))。`addr_of_mut` は dereferenceability
217//! を無視したいときに使うもので、無効化を防ぎたいときに使うものではない？
218//!
219//! ```
220//! use std::ptr::{addr_of_mut, NonNull};
221//!
222//! # struct Foo(u32, u32);
223//! # let mut foo = Foo(0, 0);
224//! let mut foo_nonnull = NonNull::from(&mut foo);
225//! let foo0_mut = unsafe { &mut (*foo_nonnull.as_ptr()).0 };
226//! let foo1_mut = unsafe { &mut (*foo_nonnull.as_ptr()).1 };
227//! *foo0_mut = 10;
228//! *foo1_mut = 20;
229//! assert_eq!((foo.0, foo.1), (10, 20)); // ok
230//! ```

232//! ### 可変な参照を返す例
233//!
234//! 上記を踏まえ、`marker::Mut<'a>` などを含めた `FooRef<BorrowType>` を考える。
235//!
236//! | `BorrowType` | like ... |
237//! |---|---|
238//! | `marker::Owned` | `Boxed<Foo>` |
239//! | `marker::Dying` | `Boxed<Foo>` (\*1) |
240//! | `marker::Immut<'a>` | `&'a Foo` |
241//! | `marker::Mut<'a>` | `&'a mut Foo` |
242//! | `marker::SndMut<'a>` | `&'a mut Foo` (\*2) |
243//!
244//! (\*1): `drop` 相当のメソッドを提供する。
245//! (\*2): `foo.0` (fst) に関しては不変参照、`foo.1` (snd) に関しては可変参照を公開する。
246//!
247//! ```ignore
248//! use std::{marker::PhantomData, ptr::NonNull};
249//!
250//! struct Foo(u32, u32);
251//! struct FooRef<BorrowType> {
252//!     foo: NonNull<Foo>,
253//!     _marker: PhantomData<BorrowType>,
254//! }
255//!
256//! todo!();
257//! ```

277use std::{
278    marker::PhantomData,
279    ptr::{self, NonNull},
280};
281
282pub struct Foo {
283    key: u32,
284    val: u32,
285}
286pub struct FooRef<BorrowType> {
287    foo: NonNull<Foo>,
288    _marker: PhantomData<BorrowType>,
289}
290
291pub mod marker {
292    use std::marker::PhantomData;
293
294    pub enum Owned {}
295    pub enum Dying {}
296    pub enum DormantMut {}
297    pub struct Immut<'a>(PhantomData<&'a ()>);
298    pub struct Mut<'a>(PhantomData<&'a ()>);
299    pub struct ValMut<'a>(PhantomData<&'a ()>);
300}
301
302impl Foo {
303    pub fn new() -> Box<Self> { Box::new(Self { key: 0, val: 0 }) }
304}
305
306impl FooRef<marker::Owned> {
307    pub fn new_foo() -> Self {
308        Self {
309            foo: NonNull::from(Box::leak(Foo::new())),
310            _marker: PhantomData,
311        }
312    }
313    pub fn borrow_mut(&mut self) -> FooRef<marker::Mut<'_>> {
314        FooRef { foo: self.foo, _marker: PhantomData }
315    }
316    pub fn borrow_valmut(&mut self) -> FooRef<marker::ValMut<'_>> {
317        FooRef { foo: self.foo, _marker: PhantomData }
318    }
319    pub fn into_dying(self) -> FooRef<marker::Dying> {
320        FooRef { foo: self.foo, _marker: PhantomData }
321    }
322}
323
324impl<BorrowType> FooRef<BorrowType> {
325    pub fn reborrow(&self) -> FooRef<marker::Immut<'_>> {
326        FooRef { foo: self.foo, _marker: PhantomData }
327    }
328    fn as_ptr(this: &Self) -> *mut Foo { this.foo.as_ptr() }
329}
330
331impl Copy for FooRef<marker::Immut<'_>> {}
332impl Clone for FooRef<marker::Immut<'_>> {
333    fn clone(&self) -> Self { *self }
334}
335
336impl<'a> FooRef<marker::Mut<'a>> {
337    pub unsafe fn reborrow_mut(&mut self) -> FooRef<marker::Mut<'_>> {
338        FooRef { foo: self.foo, _marker: PhantomData }
339    }
340    pub fn dormant(&self) -> FooRef<marker::DormantMut> {
341        FooRef { foo: self.foo, _marker: PhantomData }
342    }
343    fn as_mut(&mut self) -> &mut Foo {
344        let ptr = Self::as_ptr(self);
345        unsafe { &mut *ptr }
346    }
347    pub fn key_mut(&mut self) -> &mut u32 { &mut self.as_mut().key }
348    pub fn val_mut(&mut self) -> &mut u32 { &mut self.as_mut().val }
349}
350
351impl<'a> FooRef<marker::ValMut<'a>> {
352    pub fn into_key_valmut(mut self) -> (&'a u32, &'a mut u32) {
353        let ptr = Self::as_ptr(&mut self);
354        let key = unsafe { &*ptr::addr_of!((*ptr).key) };
355        let val = unsafe { &mut *ptr::addr_of_mut!((*ptr).val) };
356        (key, val)
357    }
358}
359
360impl FooRef<marker::DormantMut> {
361    pub unsafe fn awaken<'a>(self) -> FooRef<marker::Mut<'a>> {
362        FooRef { foo: self.foo, _marker: PhantomData }
363    }
364}
365
366impl FooRef<marker::Dying> {
367    pub fn drop(self) { unsafe { drop(Box::from_raw(Self::as_ptr(&self))) } }
368}

370#[test]
371fn test_foo_ref() {
372    let mut foo_ref = FooRef::new_foo();
373
374    let mut foo_ref_mut_1 = foo_ref.borrow_mut();
375    let key_mut = foo_ref_mut_1.key_mut();
376    assert_eq!(*key_mut, 0);
377    *key_mut += 10;
378    assert_eq!(*key_mut, 10);
379    let val_mut = foo_ref_mut_1.val_mut();
380    assert_eq!(*val_mut, 0);
381    *val_mut += 100;
382    assert_eq!(*val_mut, 100);
383
384    let mut foo_ref_mut_2 = unsafe { foo_ref_mut_1.reborrow_mut() };
385    let key_mut = foo_ref_mut_2.key_mut();
386    assert_eq!(*key_mut, 10);
387    *key_mut += 10;
388    assert_eq!(*key_mut, 20);
389    let val_mut = foo_ref_mut_2.val_mut();
390    assert_eq!(*val_mut, 100);
391    *val_mut += 100;
392    assert_eq!(*val_mut, 200);
393
394    let foo_dormant = foo_ref_mut_2.dormant();
395
396    let foo_ref_kvm = foo_ref.borrow_valmut();
397    let (key, val) = foo_ref_kvm.into_key_valmut();
398    assert_eq!(*key, 20);
399    assert_eq!(*val, 200);
400    *val += 1;
401    assert_eq!(*key, 20);
402    assert_eq!(*val, 201);
403
404    let mut foo_ref_mut_3 = unsafe { foo_dormant.awaken() };
405    let key_mut = foo_ref_mut_3.key_mut();
406    assert_eq!(*key_mut, 20);
407    *key_mut += 10;
408    assert_eq!(*key_mut, 30);
409    let val_mut = foo_ref_mut_3.val_mut();
410    assert_eq!(*val_mut, 201);
411    *val_mut += 100;
412    assert_eq!(*val_mut, 301);
413
414    // *val += 1; // UB
415
416    foo_ref.into_dying().drop();
417}

420fn ptr_read() {
421    struct Foo(u32, u32);
422
423    let mut foo = Foo(0, 0);
424
425    // {
426    //     let foo_ptr_fst: *mut _ = &mut foo;
427    //     let foo_ptr_snd: *mut _ = &mut foo;
428    //     // unsafe { (*foo_ptr_fst).0 = 10 }; // UB
429    // }
430
431    // {
432    //     let mut foo_nonnull = NonNull::from(&mut foo);
433    //     let foo_mut_fst = unsafe { &mut foo_nonnull.as_mut().0 };
434    //     let foo_mut_snd = unsafe { &mut foo_nonnull.as_mut().1 };
435    //     *foo_mut_fst = 10; // UB
436    // }
437
438    {
439        let foo_nonnull = NonNull::from(&mut foo);
440        let foo_ptr_1 = foo_nonnull.as_ptr();
441        let foo_ptr_2 = foo_nonnull.as_ptr();
442        unsafe { (*foo_ptr_1).0 += 1 };
443        unsafe { (*foo_ptr_1).1 += 2 };
444        unsafe { (*foo_ptr_2).0 += 10 };
445        unsafe { (*foo_ptr_2).1 += 20 };
446        unsafe { (*foo_ptr_1).0 += 10 };
447        unsafe { (*foo_ptr_1).1 += 10 };
448        unsafe { (*foo_ptr_2).0 -= 10 };
449        unsafe { (*foo_ptr_2).1 -= 10 };
450        assert_eq!((foo.0, foo.1), (11, 22)); // ok
451
452        unsafe { (*foo_ptr_1).0 }; // ok
453        foo.0 += 100; // invalidates `foo_ptr_1` and `foo_nonnull`
454        // unsafe { (*foo_ptr_1).0 }; // UB
455    }

459        let foo_ptr_1 = foo_nonnull.as_ptr();
