1//! 例：handle。
2//!
3//! ## Samples
4//!
5//! 以下のような API を提供するクラスを考える。
6//!
7//! - $`S \gets \varnothing`$ で初期化。
8//! - $`i = |S|`$ として $`S \gets S\cup (i, i)`$ で更新。
9//! - $`(i, j)`$ を受け取る。$`(i, j')\in S`$ として $`S \gets (S\smallsetminus (i, j'))\cup (i, j)`$ で更新。
10//! - $`i`$ を受け取る。$`\angled{i, f(i), f(f(i)), \dots}`$ を返す。
11//!     - ただし、$`{}^{\exists!}j.\: (i, j)\in S`$ のとき $`f(i) = j`$ とする。
12//!
13//! 当然 `Vec<usize>` で実現できるが、ここではポインタを扱う練習として、$`i`$ 番目のノードから
14//! $`j`$ 番目のノードに対するポインタを持つような実装を考える。
15//!
16//! ```
17//! use std::ptr::NonNull;
18//!
19//! // --- implementations ---
20//! pub struct Node {
21//!     val: usize,
22//!     mapsto: NonNull<Node>,
23//! }
24//! #[derive(Clone, Copy)]
25//! pub struct Handle {
26//!     node: NonNull<Node>,
27//! }
28//! pub struct Graph {
29//!     nodes: Vec<NonNull<Node>>,
30//! }
31//!
32//! impl Node {
33//!     pub fn new(val: usize) -> NonNull<Self> {
34//!         let node = Box::new(Node { val, mapsto: NonNull::dangling() });
35//!         let ptr = NonNull::from(Box::leak(node));
36//!         unsafe { (*ptr.as_ptr()).mapsto = ptr };
37//!         ptr
38//!     }
39//! }
40//!
41//! impl Handle {
42//!     fn new(node: NonNull<Node>) -> Self { Self { node } }
43//!     pub fn mapsto(&mut self, other: Self) {
44//!         unsafe { (*self.node.as_ptr()).mapsto = other.node }
45//!     }
46//!     pub fn walk(&self) -> impl Iterator<Item = usize> {
47//!         std::iter::successors(Some(self.node), |&ptr| unsafe {
48//!             Some((*ptr.as_ptr()).mapsto)
49//!         })
50//!         .map(|ptr| unsafe { (*ptr.as_ptr()).val })
51//!     }
52//! }
53//!
54//! impl Graph {
55//!     pub fn new() -> Self { Self { nodes: vec![] } }
56//!     pub fn push(&mut self) -> Handle {
57//!         let val = self.nodes.len();
58//!         let ptr = Node::new(val);
59//!         self.nodes.push(ptr);
60//!         Handle::new(ptr)
61//!     }
62//! }
63//!
64//! impl Drop for Graph {
65//!     fn drop(&mut self) {
66//!         for ptr in self.nodes.drain(..) {
67//!             unsafe { drop(Box::from_raw(ptr.as_ptr())) };
68//!         }
69//!     }
70//! }
71//!
72//! // --- tests ---
73//! let mut graph = Graph::new();
74//! let mut x0 = graph.push();
75//! let mut x1 = graph.push();
76//! let mut x2 = graph.push();
77//! let mut x3 = graph.push();
78//! let mut x4 = graph.push();
79//!
80//! x0.mapsto(x1);
81//! x1.mapsto(x3);
82//! x2.mapsto(x1);
83//! x3.mapsto(x1);
84//! x4.mapsto(x2);
85//!
86//! // 0 -> 1 <> 3
87//! //      ^
88//! //      2 <- 4
89//!
90//! let a: Vec<_> = x0.walk().take(10).collect();
91//! assert_eq!(a, [0, 1, 3, 1, 3, 1, 3, 1, 3, 1]);
92//!
93//! x3.mapsto(x4);
94//! x4.mapsto(x2);
95//!
96//! // 0 -> 1 -> 3
97//! //      ^    v
98//! //      2 <- 4
99//!
100//! let a: Vec<_> = x0.walk().take(10).collect();
101//! assert_eq!(a, [0, 1, 3, 4, 2, 1, 3, 4, 2, 1]);
102//!
103//! x0.mapsto(x0); // self loop
104//!
105//! let a: Vec<_> = x0.walk().take(10).collect();
106//! assert_eq!(a, [0; 10]);
107//! ```
108//!
109//! ## Notes
110//!
111//! 「node を作成し、handle を返す」「handle を渡し、構造を更新する」が基本的な方針になると思われる。
112//! より複雑な例においては「更新の際に、別の handle（が持っている pointer が指す node）を無効化する」
113//! ということも起きうるので、注意する必要がある。
114//!
115//! これが問題なく書けるのであれば、Fibonacci heap なども書けそうな気がするが果たして？
116//!
117//! おそらくは次のような構造で書けると思う (cf. [CS166 (1186), Fibonacci Heaps](https://web.stanford.edu/class/archive/cs/cs166/cs166.1186/lectures/09/Slides09.pdf#page=208))。
118//!
119//! ```
120//! use std::ptr::NonNull;
121//!
122//! pub struct FibHeap<T: Ord> {
123//!     roots: Vec<FibHeapNode<T>>,
124//!     max: Option<T>,
125//! }
126//! struct FibHeapNode<T> {
127//!     val: T,
128//!     parent: Option<NonNull<FibHeapNode<T>>>,
129//!     neighbor: (NonNull<FibHeapNode<T>>, NonNull<FibHeapNode<T>>),
130//!     any_child: Option<NonNull<FibHeapNode<T>>>,
131//!     order: usize,
132//!     cut: bool,
133//! }
134//! pub struct FibHeapHandle<T> {
135//!     node: NonNull<FibHeapNode<T>>,
136//! }
137//!
138//! impl<T: Ord> FibHeap<T> {
139//!     pub fn new() -> Self { todo!() }
140//!     pub fn push(&mut self, elt: T) -> FibHeapHandle<T> { todo!() }
141//!     pub fn pop(&mut self) -> Option<T> { todo!() }
142//!     pub fn meld(&mut self, other: Self) { todo!() }
143//! }
144//! impl<T: Ord> FibHeapHandle<T> {
145//!     pub unsafe fn urge(&mut self, elt: T) -> bool { todo!() }
146//! }
147//! ```
148//!
149//! 優先度を変えるインターフェースとしては [`std::collections::binary_heap::PeekMut`]
150//! のようなものも存在しているが、Fibonacci heap を使いたい状況では、最大値に限らない要素を（検索含め）
151//! 償却定数時間で行える必要があり、handle 経由で行う必要があると思われる。当然、deterministic
152//! に行いたいので hash map は使いたくない。
