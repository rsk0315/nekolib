1use std::ptr::NonNull;
2
3struct ListNode {
4    val: i32,
5    next: Option<NonNull<ListNode>>,
6}
7
8pub struct List {
9    first_last: Option<(NonNull<ListNode>, NonNull<ListNode>)>,
10}
11
12impl ListNode {
13    fn new(val: i32) -> NonNull<Self> {
14        NonNull::from(Box::leak(Box::new(Self { val, next: None })))
15    }
16}
17
18impl List {
19    pub fn new() -> Self { Self { first_last: None } }
20    pub fn push(&mut self, elt: i32) {
21        let node = ListNode::new(elt);
22        if let Some((_, last)) = &mut self.first_last {
23            unsafe { (*last.as_ptr()).next = Some(node) };
24            *last = node;
25        } else {
26            self.first_last = Some((node, node));
27        }
28    }
29    pub fn iter(&self) -> impl Iterator<Item = i32> + '_ {
30        std::iter::successors(
31            self.first_last.map(|(first, _)| first),
32            |node| unsafe { (*node.as_ptr()).next },
33        )
34        .map(|node| unsafe { (*node.as_ptr()).val })
35    }
36}
37
38impl Drop for List {
39    fn drop(&mut self) {
40        if let Some((first, _)) = self.first_last {
41            let mut next = Some(first);
42            while let Some(node) = next {
43                next = unsafe { (*node.as_ptr()).next };
44                unsafe { drop(Box::from_raw(node.as_ptr())) };
45            }
46        }
47    }
48}
49
50#[test]
51fn sanity_check() {
52    let mut list = List::new();
53    assert!(list.iter().eq([]));
54
55    list.push(1);
56    list.push(2);
57    assert!(list.iter().eq([1, 2]));
58}
